49//------------------------------------------------------------------------------
53//
54//
55// Flow of information between widget, model view and visual pipe.
56//
57// MODEL VIEW
58// slice id
59// | Original contour <---------------------------------------- initialization
60// | centres only
61// WIDGET | | ^
62// V V |
63// Interaction----------> Splines slice id |
64// | | / \ |
65// | |<--------- ------------ |
66// | | \ VISUAL PIPE 2D |
67// V V \ |
68// rot. handle <---- Spline(zeta) -------> Transforms ---> contour --------> contour segments -->
69// scaling handles <---- ---> cut planes -----> |
70// ^ ---> contour axes |
71// | V
72// <-----------------------------------------------------------------------------
