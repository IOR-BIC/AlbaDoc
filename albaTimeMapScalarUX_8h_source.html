310//-------------------------------------------------------------------------
311template <class T>
312typename albaTimeMapScalar<T>::TimeMapScalars::iterator albaTimeMapScalar<T>::FindNearestItem(albaTimeStamp t)
313//-------------------------------------------------------------------------
314{
320 albaTimeMapScalar<T>::TimeMapScalars::iterator lowIt;
321 albaTimeMapScalar<T>::TimeMapScalars::iterator upIt;
322
323 lowIt = FindItemBefore(t); // find first item < t
324 upIt = m_TimeMap.lower_bound(t); // find first item >= t
325
326 if(lowIt != m_TimeMap.end())
327 {
328 if(upIt != m_TimeMap.end())
329 {
330 if(fabs(lowIt->first - t) < fabs(upIt->first - t))
331 {
332 return lowIt;
333 }
334 else
335 {
336 return upIt;
337 }
338 }
339 return lowIt;
340 }
341 else if(upIt != m_TimeMap.end())
342 {
343 return upIt;
344 }
345 return --upIt;
346}
