139inline albaICPUtility::RegResult albaICPUtility::SVD_Registration(vnl_matrix<double>& MS, vnl_matrix<double>& DS)
140//----------------------------------------------------------------------------
141{
143
144 vnl_vector_fixed<double,3> MS_centroid=Centroid(MS);
145 vnl_vector_fixed<double,3> DS_centroid=Centroid(DS);
146
147 //vcl_cout << "Dentro SVD il centroide di MS vale: " << MS_centroid << '\n';
148 //vcl_cout << "Dentro SVD il centroide di DS vale: " << DS_centroid << '\n';
149
150 Neg_translation(MS, MS_centroid);
151 Neg_translation(DS, DS_centroid);
152
153 vnl_matrix<double> MSt=MS.transpose();
154 vnl_matrix_fixed<double,3,3> Z=MSt*DS; //MS deve essere 3xn mentre DS una nx3
155
156 vnl_svd<double> s_v_d(Z);
157
158 vnl_matrix<double> V=s_v_d.V();
159 V.inplace_transpose();
160
161 vnl_vector_fixed<double,3> temp(1,1,vnl_determinant(s_v_d.U()*V));
162 vnl_diag_matrix<double> diag(temp);
163
164 albaICPUtility::RegResult registrazione_S_V_D;
165 registrazione_S_V_D.R=s_v_d.U()*diag*V;
166 registrazione_S_V_D.t=MS_centroid-(registrazione_S_V_D.R)*DS_centroid;
167
168 //vcl_cout << "Entro SVD R finale vale: \n" << registrazione_S_V_D.R << '\n';
169 //vcl_cout << "Entro SVD t finale vale: \n" << registrazione_S_V_D.t << '\n';
170
171 Pos_translation(MS,MS_centroid); //Se le matrici sono passate per riferimento questo serve! se no commentare
172 Pos_translation(DS,DS_centroid); //Se le matrici sono passate per riferimento questo serve! se no commentare
173
174 return registrazione_S_V_D;
175}

201inline vnl_matrix_fixed<double,3,3> albaICPUtility::Theta2R(const vnl_vector_fixed<double,3>& th)
202//----------------------------------------------------------------------------
203{
204 double fi=th.magnitude();
205 double sinc=sin(fi)/fi; //sin accept rediants, such as MATLAB
206 double cosc=(1-cos(fi))/(fi*fi);
207
208 vnl_matrix_fixed<double,3,3> A;
209 A(0,0)=0; A(0,1)=-th(2)*sinc; A(0,2)=th(1)*sinc;
210 A(1,0)=th(2)*sinc; A(1,1)=0; A(1,2)=-th(0)*sinc;
211 A(2,0)=-th(1)*sinc; A(2,1)=th(0)*sinc; A(2,2)=0;
212
213 vnl_diag_matrix<double> first_term(3, cos(fi));
214
215 vnl_matrix_fixed<double,3,3> third_term;
216 for(int i=0; i<3; i++) third_term.set_row(i, cosc*th(i)*th);
217
218 vnl_matrix_fixed<double,3,3> R=first_term+A.as_matrix()+third_term.as_matrix();
219 return R;
220}

250//----------------------------------------------------------------------------
251inline albaICPUtility::RegResult albaICPUtility::ICP_vtkPointLocator(vnl_matrix<double>& MS, vnl_matrix<double> & DS, int cycle_lim)
252//----------------------------------------------------------------------------
253{
254 //-----------Contruzione Point Locator----------------------
255 vtkPoints *MS_vtkpoints = vtkPoints::New();
256 MS_vtkpoints->SetNumberOfPoints(MS.rows());
257 for (int i = 0; i < MS.rows(); i++) {
258 MS_vtkpoints->InsertPoint(i, MS(i,0), MS(i,1), MS(i,2));
259 }
260
261 vtkPolyData *MS_poly = vtkPolyData::New();
262 MS_poly->SetPoints(MS_vtkpoints);
263 MS_poly->Modified();
264
265 //vcl_cout << "Start building PointLocator" <<'\n';
266 vtkPointLocator *p_locator = vtkPointLocator::New();
267 p_locator->SetDataSet(MS_poly);
268 p_locator->BuildLocator();
269 //vcl_cout << "Finish building PointLocator" <<'\n';
270 //-----------------------------------------------------------
271
272 DS.inplace_transpose(); //Dato che di DS non me ne faccio niente dritto, transpongo il sorgente
273
274 const int DSrows = DS.rows(); //ora le righe sono 3...
275 const int DScols = DS.columns(); //...e le colonne n
276
277 double costov=1E+6;
278 double coston=1E+5;
279
280 double b=2E-5;
281 const double a=1+1E-5;
282
283 vnl_vector_fixed<double,3> th(1E-5,1E-5,1E-5);
284 vnl_vector_fixed<double,3> t(1E-5,1E-5,1E-5);
285
286 double test=1; double test_lim=1E-5;//orig:1E-5
287 double err=100; double err_lim=1E-2;//orig:1E-2
288 double var_err=1E+3; double var_err_lim=1E-12;
289 int cycle=0; //int cycle_lim=28;//Usualmente 28 NOW PASSED FROM THE USER INTERFACE
290
291 //Initialization of partial result needed to compute dy
292 vnl_matrix<double> dy(DSrows, DScols);
293 vnl_vector_fixed<double,3> th_a; //es: th.*[a 1 1]
294 vnl_vector_fixed<double,3> t_a; //es: t.*[a 1 1]
295 vnl_matrix<double> R_DS(DSrows, DScols); //es: Theta2R(th)*DS
296 vnl_matrix<double> translation(DSrows, DScols); //es: t*ones(1,n)
297 vnl_matrix<double> DS_move(DSrows, DScols); //es: Theta2R(th)*DS+t*ones(1,n)
298 vnl_matrix<double> S(DSrows*DScols, 6);
299 vnl_matrix<double> St(6,DSrows*DScols);
300
301 vnl_matrix<double> closest_pts(DSrows, DScols);
302 vnl_matrix<double> em(DSrows, DScols);
303 vnl_vector<double> e(DSrows*DScols);
304 vnl_vector_fixed<double,6> dv;
305
306 while(test>test_lim && err>err_lim && var_err > var_err_lim && cycle < cycle_lim){
307 cycle++;
308 //vcl_cout << "ciclo numero: \n" << cycle << '\n';
309 if(cycle>8) b=1E-5; //era 8 metto 1
310
311 //Caricamente strutture parziali
312 R_DS=Theta2R(th)*DS;
313 for (int i=0; i< DScols; i++) translation.set_column(i,t);
314 DS_move=R_DS+translation;
315
316
317 for(int j=0; j<3; j++){
318 th_a=th;
319 th_a.put(j, th[j]*a);
320
321
322 dy=( (Theta2R(th_a)*DS)+translation-DS_move ) / ( th[j]*b );
323 S.set_column(j,dy.data_block());
324 }
325
326 for(int j=0; j<3; j++){
327 t_a=t;
328 t_a.put(j, t[j]*a);
329
330 for (int i=0; i< DScols; i++) translation.set_column(i,t_a);
331
332 dy=( R_DS+translation-DS_move ) / ( t[j]*b );
333 S.set_column(j+3,dy.data_block());
334 }
335
336 FindPointLocator(MS_vtkpoints, p_locator, DS_move.data_block(), DScols, closest_pts);
337
338 em=closest_pts-DS_move;
339 e.set(em.data_block());
340
341 //Aggiorno condizioni di ciclo
342 coston=e.squared_magnitude();
343 test=abs(coston-costov)/costov;
344 costov=coston;
345 err = sqrt( coston / (3*DScols) );
346 var_err = coston / ((3*DScols)-6);
347
348 //Aggiorno i parametri di trasformazione
349 St=S.transpose();
350 vnl_matrix_inverse<double> inv_St_S(St*S);
351 vnl_matrix<double> dv_mat_temp=inv_St_S*St;
352 dv=dv_mat_temp*e;
353 th +=dv.extract(3,0);
354 t +=dv.extract(3,3);
355 }
356
357 DS=DS_move;
358 DS.inplace_transpose();
359
360 albaICPUtility::RegResult res;
361 res.R=Theta2R(th);
362 res.t=t;
363 res.err= err;
364 res.var = var_err;
365 res.test=test;
366 //vcl_cout << "var finale vale: "<< res.var << var_err << '\n';
367 //vcl_cout << "test finale vale: " << res.test << test << '\n';
368 //vcl_cout << "err finale vale: " << res.err << err << '\n';
369 //vcl_cout << "th vale: \n" << th << '\n';
370 //vcl_cout << "t vale: \n" << t << '\n';
371 vnl_matrix_inverse<double> inv_St_S(St*S);
372 vnl_matrix_fixed<double,6,6> covariance_mat_temp=inv_St_S;
373 res.cov_matrix=var_err*covariance_mat_temp;
374 //vcl_cout << res.cov_matrix <<'\n';
375
376 MS_vtkpoints->Delete();
377 MS_poly->Delete();
378 p_locator->Delete();
379 return res;
380}

382inline vnl_matrix_fixed<double,3,3> albaICPUtility::Rpy2R(const vnl_vector_fixed<double,3> v)
383//----------------------------------------------------------------------------
384{
385 double pi=3.14159265358979323846;
386 double xrad=(v[0]*pi)/180;
387 double yrad=(v[1]*pi)/180;
388 double zrad=(v[2]*pi)/180;
389
390 vnl_matrix_fixed<double,3,3> rx,ry,rz,R;
391 rx(0,0)=1 ;rx(0,1)=0 ;rx(0,2)=0 ;
392 rx(1,0)=0 ;rx(1,1)=cos(xrad) ;rx(1,2)=-sin(xrad) ;
393 rx(2,0)=0 ;rx(2,1)=sin(xrad) ;rx(2,2)=cos(xrad) ;
394
395 ry(0,0)=cos(yrad) ;ry(0,1)=0 ;ry(0,2)=sin(yrad) ;
396 ry(1,0)=0 ;ry(1,1)=1 ;ry(1,2)=0 ;
397 ry(2,0)=-sin(yrad) ;ry(2,1)=0 ;ry(2,2)=cos(yrad) ;
398
399 rz(0,0)=cos(zrad) ;rz(0,1)=-sin(zrad) ;rz(0,2)=0 ;
400 rz(1,0)=sin(zrad) ;rz(1,1)=cos(zrad) ;rz(1,2)=0 ;
401 rz(2,0)=0 ;rz(2,1)=0 ;rz(2,2)=1 ;
402
403 R=rz*ry*rx;
404
405 return R;
406}
