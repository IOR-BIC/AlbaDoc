287//------------------------------------------------------------------------------
288// Add set of id's to the edge mapping table
289// This is a POINT_TO_EDGE mapping
290// This maps output point idout to input edge (id0,id1)
291inline void vtkALBAMeshCutter::AddMapping(vtkIdType idout, const Edge& edge, double lambda)
292//------------------------------------------------------------------------------
293{
294 EdgeMappingType em = {idout, edge.id0, edge.id1, lambda, POINT_TO_EDGE} ;
295 EdgeMapping.push_back(em);
296
297 InvEdgeMapping.insert(InvEdgeMappingType::value_type(GetEdgeKey(edge), EdgeMapping.size() - 1));
298}
299
300//------------------------------------------------------------------------------
301// Add set of id's to the edge mapping table
302// this is a POINT_TO_POINT mapping
303// This maps output point idout to single input point id0
304inline void vtkALBAMeshCutter::AddMapping(vtkIdType idout, vtkIdType id0, double lambda)
305//------------------------------------------------------------------------------
306{
307 EdgeMappingType em = {idout, id0, undefinedId, lambda, POINT_TO_POINT} ;
308 EdgeMapping.push_back(em) ;
309
310 InvEdgeMapping.insert(InvEdgeMappingType::value_type(id0, EdgeMapping.size() - 1));
311}

313//------------------------------------------------------------------------
314//computes a unique key for the given edge
315inline unsigned long long vtkALBAMeshCutter::GetEdgeKey(vtkIdType id0, vtkIdType id1) const
316//------------------------------------------------------------------------
317{
318 unsigned long long key;
319 unsigned long* pkey = (unsigned long*)&key;
320 if (id0 < id1) {
321 pkey[0] = id0; pkey[1] = id1;
322 }
323 else {
324 pkey[0] = id1; pkey[1] = id0;
325 }
326
327 return key;
328}

330//------------------------------------------------------------------------------
331// Find the input edge (id0,id1) corresponding to the output point idout
332// This returns true if the output point is in the table
333inline bool vtkALBAMeshCutter::GetInputEdgeCutByPoint(vtkIdType idout, vtkIdType *id0, vtkIdType *id1, double *lambda) const
334//------------------------------------------------------------------------------
335{
336 //RELEASE NOTE: as AddMapping is called in a pair with InsertNextPoint in the current version,
337 //idout is the index into EdgeMapping, thus we do not need search for it
338 assert(idout >= 0 && idout < (int)EdgeMapping.size());
339 const EdgeMappingType& edge = EdgeMapping[idout];
340 if (edge.mtype != POINT_TO_EDGE)
341 return false;
342
343 assert(edge.idout == idout);
344
345 *id0 = edge.id0 ;
346 *id1 = edge.id1 ;
347 *lambda = edge.lambda ;
348 return true;
349}

387//------------------------------------------------------------------------------

394//------------------------------------------------------------------------------
395// Interpolate two scalar tuples
396// tupout = (1-lambda)*tup0 + lambda*tup1
397// If the data type is not float or double, we add 0.5 to ensure correct rounding when it is converted back to int.
398inline void vtkALBAMeshCutter::InterpolateScalars(double *tup0, double *tup1, double *tupout, double lambda, int ncomponents, int dattype)
399//------------------------------------------------------------------------------
400{
401 for (int j = 0 ; j < ncomponents ; j++)
402 tupout[j] = (1.0 - lambda)*tup0[j] + lambda*tup1[j] ;
403
404 if (dattype != VTK_FLOAT && dattype != VTK_DOUBLE){
405 // the real data type of the scalars is integer, so add 0.5 to ensure correct rounding when they are put back into the polydata
406 for (int j = 0 ; j < ncomponents ; j++)
407 tupout[j] += 0.5 ;
408 }
409}
