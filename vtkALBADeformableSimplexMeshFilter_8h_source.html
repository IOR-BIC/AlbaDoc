323 //---------------------------------------------------------------------------
324 // Is point x above cell (triangle only). \n
325 // 1 for above, 0 for in, -1 for below
326 //---------------------------------------------------------------------------
327 template< typename TInputMesh, typename TOutputMesh >
328 int vtkALBADeformableSimplexMeshFilter<TInputMesh, TOutputMesh >
329 ::IsPointAboveCell(vtkPolyData *polydata, int cellId, const double x[3]) const
330 {
331 // check that cell is triangle
332 vtkIdList *ids = vtkIdList::New() ;
333 polydata->GetCellPoints(cellId, ids) ;
334 int n = ids->GetNumberOfIds() ;
335 assert(n == 3) ;
336
337 // get normal of cell
338 double p0[3], p1[3], p2[3], a[3], b[3], xp[3], s[3] ;
339 polydata->GetPoint(ids->GetId(0), p0) ;
340 polydata->GetPoint(ids->GetId(1), p1) ;
341 polydata->GetPoint(ids->GetId(2), p2) ;
342 for (int j = 0 ; j < 3 ; j++){
343 a[j] = p1[j]-p0[j] ;
344 b[j] = p2[j]-p0[j] ;
345 xp[j] = x[j] - p0[j] ;
346 vtkMath::Cross(a,b,s) ;
347 }
348
349 ids->Delete() ;
350
351 // get dot product with normal and return sign
352 double z = vtkMath::Dot(xp,s) ;
353 if (z > m_MinDistanceToVessel)
354 return 1 ;
355 else if (z < -m_MinDistanceToVessel)
356 return -1 ;
357 else
358 return 0 ;
359 }

366 template< typename TInputMesh, typename TOutputMesh >
367 int vtkALBADeformableSimplexMeshFilter< TInputMesh, TOutputMesh >
368 ::IsPointInsideVessel1(const double *stentPoint, double *surfacePoint) const
369 {
370 int ptId = m_SurfacePoly->FindPoint(surfacePoint) ;
371 vtkIdList *cellNeighbours = vtkIdList::New() ;
372 m_SurfacePoly->GetPointCells(ptId, cellNeighbours) ;
373 int inside = 1 ;
374 for (int i = 0 ; i < cellNeighbours->GetNumberOfIds() && inside == 1 ; i++){
375 int cellId = cellNeighbours->GetId(i) ;
376 int aboveCell = IsPointAboveCell(m_SurfacePoly, cellId, stentPoint) ;
377 // Point is outside if it is outside with respect to any one of the cells.
378 // This should be ok as long as the vessel surface is concave.
379 if ((aboveCell == 0) || (aboveCell == 1))
380 inside = 0 ;
381 }
382
383 cellNeighbours->Delete() ;
384 return inside ;
385 }

392 template< typename TInputMesh, typename TOutputMesh >
393 int vtkALBADeformableSimplexMeshFilter< TInputMesh, TOutputMesh >
394 ::IsPointInsideVessel2(const double *stentPoint, double *surfacePoint) const
395 {
396 int ptId = m_SurfacePoly->FindPoint(surfacePoint) ;
397 double *norm = m_SurfacePoly->GetPointData()->GetNormals()->GetTuple3(ptId) ;
398 double dprod = 0.0 ;
399 for (int j = 0 ; j < 3 ; j++)
400 dprod += (surfacePoint[j] - stentPoint[j]) * norm[j] ;
401
402 if (dprod < -m_MinDistanceToVessel)
403 return 1 ;
404 else
405 return 0 ;
406 }
