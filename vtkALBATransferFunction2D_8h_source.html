36struct tfWidget {
37 char Name[32];
38
39 int GradientInterpolationOrder; // 0 - no interpolation, 1 linear
40
41 // position
42 double Ratio;
43 double Range[2][3]; // 0 - left, 1 - right, 2 - center
44
45 // value
46 double Color[3];
47 double Opacity;
48 double Diffuse;
49
50 bool Visible;
51
52 // data calculated from the above parameters
53 double LengthI[2][2]; // 1 / (Range[x][i] - Range[x][1])
54 double RatioI;
55
56 // methods
57 tfWidget() { this->Name[0] = '\0'; this->GradientInterpolationOrder = 0; this->Ratio = 1.f; this->Opacity = 1.f; this->Diffuse = 0.f; this->Color[0] = this->Color[1] = this->Color[2] = 1.f; this->Visible = true; }
58
59 void SetValueRange(double from, double to, double center) { Range[0][0] = from; Range[0][1] = to; Range[0][2] = center; }
60 void SetGradientRange(double from, double to, double center) { Range[1][0] = from; Range[1][1] = to; Range[1][2] = center; }
61
62 bool Inside(double val, double grad) const { return (val >= this->Range[0][0] && val <= this->Range[0][1] && grad >= this->Range[1][0] && grad <= this->Range[1][1]); }
63 ALBA_EXPORT double Attenuation(double val, double grad) const;
64
65 void Update();
66 };
