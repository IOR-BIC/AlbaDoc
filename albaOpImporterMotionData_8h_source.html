88//----------------------------------------------------------------------------
90 void OpRun()
91 {
92 int result = OP_RUN_CANCEL;
93 m_File = "";
94 m_Dict = "";
95
96 albaString f = albaGetOpenFile(m_FileDir,m_PgdWildc);
97 if(f != "")
98 {
99 m_File = f;
100 f = albaGetOpenFile(m_DictDir,m_DicWildc,"Open Dictionary");
101 if(f != "")
102 {
103 m_Dict = f;
104 SetDictionaryFlagOn();

107 else

113 albaEventMacro(albaEvent(this,result));

115//----------------------------------------------------------------------------
117 void OpDo()
118 {
119 // Modified by Simone Brazzale, 03/12/2010
120 assert(!m_Vme);
121
122 if (!m_TestMode)
123 {
124 wxBusyInfo wait("Please wait, working...");
125 }
126
127 albaSmartPointer<MotionReader> reader;
128 reader->SetFileName(m_File);
129 reader->SetDictionaryFileName(m_Dict);
130
131 if (GetDictionaryFlag()==1)
132 reader->DictionaryOn();
133 else
134 reader->DictionaryOff();
135
136 reader->Read();
137
138 m_Vme = reader;
139
140 wxString path, name, ext;
141 wxFileName::SplitPath(m_File,&path,&name,&ext);
142 m_Vme->SetName(name);
143
144 albaTagItem tag_Nature;
145 tag_Nature.SetName("VME_NATURE");
146 tag_Nature.SetValue("NATURAL");
147
148 m_Vme->GetTagArray()->SetTag(tag_Nature);
149
150 // Must register in order to preserve output for do/undo operation (since it is a smart pointer)
151 m_Output = m_Vme;
152 m_Vme->Register(m_Output);
153 GetLogicManager()->VmeAdd(m_Vme);
154 }
