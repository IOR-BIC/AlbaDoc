226inline int albaOpImporterRAWImages::GetVTKDataType()
227//------------------------------------------------------------------------
228{
229 const static int BIT2VTK[] = {
230 VTK_UNSIGNED_CHAR, VTK_CHAR,
231 VTK_UNSIGNED_SHORT, VTK_SHORT,
232 VTK_UNSIGNED_SHORT, VTK_SHORT,
233 VTK_UNSIGNED_CHAR, VTK_UNSIGNED_CHAR,
234 };
235
236 return BIT2VTK[2*m_Bit + m_Signed];
237}
