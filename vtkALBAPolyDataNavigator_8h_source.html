44//
45// For Developers:
46//
47// polydata->GetCellPoints() is much faster than polydata->GetCell()->GetCellIds(),
48// which has only serial access to the cells.
49//
50// Methods which use GetCellPoints() need BuildCells().
51// Methods which use GetPointCells() need BuildLinks().
52// All the point, edge and cell methods depend on at least one of these.
53// GetCellPoints() and GetPointCells() will build the links themselves if they are not already built,
54// but will NOT update existing links if they are invalid, causing a crash.
55// Therefore methods which change the polydata should finish by deleting the invalid links,
56// or carry a warning that the user should do it.
57//
58// Version: Nigel McFarlane 2.4.14

599{
600public:
601 Edge(int id0, int id1) : m_Id0(id0), m_Id1(id1) {}
602 int GetId0() const {return m_Id0 ;}
603 int GetId1() const {return m_Id1 ;}
604 void SetId0(int id0) {m_Id0 = id0 ;}
605 void SetId1(int id1) {m_Id1 = id1 ;}
606 int GetOtherId(int id) const {if (id == m_Id0) return m_Id1 ; else if (id == m_Id1) return m_Id0 ; else return -1 ;}
607 bool ContainsPoint(int id) const {return ((m_Id0 == id) || (m_Id1 == id)) ;}
608 bool IsSameDirection(const Edge& edge) const {return (m_Id0 == edge.m_Id0) && (m_Id1 == edge.m_Id1) ;}
609 bool IsReversed(const Edge& edge) const {return (m_Id0 == edge.m_Id1) && (m_Id1 == edge.m_Id0) ;}
610 bool IsJoined(const Edge& edge) const ;
611 bool IsValid() const {return ((m_Id0>=0)&&(m_Id1>=0));}
612 void Reverse() {std::swap(m_Id0,m_Id1) ;}
613 int GenerateKey() const ;
614 void PrintSelf(ostream& os, vtkIndent indent) const {os << m_Id0 << " " << m_Id1 << std::endl ;}
615 bool operator==(const Edge& edge) const ;
616 bool operator!=(const Edge& edge) const ;
617private:
618 vtkIdType m_Id0 ;
619 vtkIdType m_Id1 ;
620};
